2015-08-13 PHPзј–зЁӢ

controllerпјҡ

жҺ§еҲ¶еҷЁзұ»еҗҚеҝ…йЎ»д»ҘеӨ§еҶҷеӯ—жҜҚејҖеӨҙ
         class  User  extends  CI_Controller{}          //е»әи®®еҶҷжі•
         class  BlogUser  extends  CI_Controller{}  //е»әи®®еҶҷжі•  
 д»ҘдёӢиҝҷж ·еҶҷжҳҜж— ж•Ҳзҡ„
         class  blog  extends  CI_Controller{}
         class  blogUser  extends  CI_Controller{}

 жҺ§еҲ¶еҷЁзұ»зҡ„ж–№жі•жІЎжңүиҰҒжұӮ

 жҺ§еҲ¶еҷЁж–Үд»¶еҗҚд№ҹжІЎжңүиҰҒжұӮпјҢдҪҶжҳҜе»әи®®е…ЁйғЁе°ҸеҶҷпјҢдё”жіЁж„Ҹж–Үд»¶еҗҚе’Ңзұ»еҗҚеҗҢдёҖдёӘеҚ•иҜҚ
 controller/blog.php гҖҒcontroller/bloguser.php

model

жЁЎеһӢзұ»еҗҚйҰ–еӯ—жҜҚд№ҹеҝ…йЎ»еӨ§еҶҷпјҢе…¶д»–е°ҸеҶҷ
дҫӢеҰӮпјҡ
class  Blog_model  extends  CI_Model {    //class BlogModel extends  CI_Modelд№ҹеҸҜ
    function __construct(){
        parent::__construct();
    }
}

иҪҪе…Ҙmodelд»ҘеҸҠдҪҝз”Ёmodel
$this->load->model('Blog_model');    //modelзұ»еҗҚ  $this->load->model('BlogModel'); 
$this->Blog_model->function();         //иҪҪе…Ҙд№ӢеҗҺеҸҜзӣҙжҺҘйҖҡиҝҮзұ»еҗҚи®ҝй—®е…¶ж–№жі•$this->BlogModel->function();

жЁЎеһӢзҡ„ж–Үд»¶еҗҚжІЎжңүиҰҒжұӮпјҢдҪҶжҳҜе»әи®®е…ЁйғЁе°ҸеҶҷ,дё”жіЁж„Ҹж–Үд»¶еҗҚе’Ңзұ»еҗҚеҗҢдёҖдёӘеҚ•иҜҚ 
дҫӢеҰӮпјҡmodel/blog_model.php   //       model/blogmodel.php

view

ж–Үд»¶еҗҚжҳҜ views/viewname.php
иҪҪе…Ҙи§Ҷеӣҫ $this->load->view("viewname");
жҖ»д№Ӣпјҡж–Үд»¶еҗҚе’ҢиҪҪе…ҘдёҖиҮҙеҚіеҸҜпјҢеӨ§е°ҸеҶҷжІЎжңүзү№еҲ«иҰҒжұӮ
дҫӢеҰӮпјҡж–Үд»¶еҗҚдёәviews/viewName.php  иҪҪе…Ҙж—¶$this->load->view("viewName");

з”Ёеӯҗж–Үд»¶еӨ№еӯҳеӮЁи§Ҷеӣҫ
views/user/bloguser.php   иҪҪе…Ҙж—¶$this->load->view("user/bloguser");



жҖ»з»“пјҢеҜ№дәҺжЁЎеһӢе’ҢжҺ§еҲ¶еҷЁпјҢзұ»еҗҚзҡ„йҰ–еӯ—жҜҚеҝ…йЎ»еӨ§еҶҷпјҢе…¶д»–еӯ—жҜҚж— иҰҒжұӮпјҢеҸҜж №жҚ®дёӘдәәд№ жғҜ
          ж–Үд»¶еҗҚжІЎжңүзү№еҲ«иҰҒжұӮпјҢе»әи®®йҰ–еӯ—жҜҚе°ҸеҶҷжҲ–е…Ёе°ҸеҶҷпјҢеҗҚеӯ—дёҺзұ»еҗҚдёҖиҮҙпјҲйҷӨеӨ§е°ҸеҶҷеҸҜдёҚдёҖиҮҙпјүиҪҪе…Ҙзҡ„жЁЎеһӢе’ҢжЁЎеһӢзұ»еҗҚдёҖиҮҙ

          еҜ№дәҺи§ҶеӣҫпјҢж–Үд»¶еҗҚе’ҢиҪҪе…Ҙж—¶дёҖиҮҙеҚіеҸҜпјҢиҪҪе…Ҙж—¶дёҚеёҰеҗҺзјҖ 

  еӣ дёәmodelе’ҢcontrollerеҰӮжһңеҫҲеӨҡпјҢиҪҪе…Ҙmodelж—¶дёҚеҸҜиғҪдёҖдёӘдёҖдёӘжү“ејҖжҹҘзңӢе…¶зұ»еҗҚпјҢдҪҶжҳҜеҸҜд»Ҙеҝ«йҖҹж №жҚ®ж–Үд»¶еҗҚзЎ®е®ҡе…¶зұ»еҗҚ

  йӮЈе°ұиҰҒжұӮеҜ№ж–Үд»¶зҡ„е‘ҪеҗҚе’Ңзұ»зҡ„е‘ҪеҗҚиҰҒжңүдёҖдёӘдёҖиҮҙзҡ„и§„иҢғ

жҲ‘зҡ„д№ жғҜпјҲдҝқжҢҒпјү
  гҖҗ1гҖ‘controller,зұ»еҗҚUserжҲ–UserBlogпјҢж–Үд»¶еҗҚuser.phpжҲ–userblog.php
  гҖҗ2гҖ‘modelпјҢзұ»еҗҚUserModelжҲ–UserBlogModel пјҢж–Үд»¶еҗҚusermodel.phpжҲ–userblogmodel.php
   иҪҪе…Ҙж–№жі•$this->load->model("UserModel");
   з”ЁеңЁжһ„йҖ ж–№жі•йҮҢеҰӮдёӢпјҡ
   class BlogModel  extends  CI_Controller{
       function __construct(){
            parent::__construct();
            $this->load->model('UserModel'); 
        }
    }
  гҖҗ3гҖ‘viewпјҢж–Үд»¶еҗҚuserblog.php жҲ–user/blog.php  иҪҪе…Ҙж–№жі•$this->load->view("userblog");жҲ–
